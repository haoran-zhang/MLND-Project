graph_2=tf.Graph()

# Initiate parameters

with graph_2.as_default():
    
    # Place_holder for 1 image with 5 labels input
    x = tf.placeholder("float32", shape=[None, 1024])
    y_1 = tf.placeholder("float32", shape=[None, 11])
    y_2 = tf.placeholder("float32", shape=[None, 11])
    y_3 = tf.placeholder("float32", shape=[None, 11])
    y_4 = tf.placeholder("float32", shape=[None, 11])
    y_5 = tf.placeholder("float32", shape=[None, 11])

    # define this four functions, in order to fix some parameters so as to simplify the
    # initiate parameter process
    def bias_variable(shape):
        return tf.Variable(tf.random_uniform(shape, minval=-0.1, maxval=0.1))

    def weight_variable(shape,name):
        return tf.Variable(tf.random_uniform(shape=shape, minval=-0.1, maxval=0.1),name=name)

    def conv2d(x, W):
        return tf.nn.conv2d(x, W, strides=[1, 1, 1, 1], padding='VALID')

    def max_pool_2x2(x):
        return tf.nn.max_pool(x, ksize=[1, 2, 2, 1],
                              strides=[1, 2, 2, 1], padding='SAME')

    # First convolution layer, transfer 1 feature map to 32 feature maps
    x_image = tf.reshape(x, [-1, 32, 32, 1])

    W_conv1 = weight_variable([5, 5, 1, 16],'W_conv1')
    b_conv1 = bias_variable([16])
    h_conv1 = tf.nn.relu(conv2d(x_image, W_conv1) + b_conv1)

    h_pool1 = max_pool_2x2(h_conv1)

    # Second convolution layer, transfer 16 feature maps to 64 feature maps
    W_conv2 = weight_variable([5, 5, 16, 64],'W_conv2')
    b_conv2 = bias_variable([64])

    h_conv2 = tf.nn.relu(conv2d(h_pool1, W_conv2) + b_conv2)
    h_pool2_pre = max_pool_2x2(h_conv2)

    # First dropout layer
    keep_prob = tf.placeholder("float")
    h_pool2 = tf.nn.dropout(h_pool2_pre, keep_prob)
    
    """
    The two convolutional layers, followed by a little more complex inception layer, 
    which containing 1-1 conv net, 3-3 conv net after 1-1 conv net, and 3-3 max-pooling after
    1-1 conv net. Then, this three parts was concated and pass to another layer. 
    More details are in the report.
    """
    
    # INCEPTION LAYER
    # 1-1 convolution net
    W_11_1 = weight_variable([1, 1, 64, 64],'W_11_1')
    b_11_1 = bias_variable([64])
    h_11_conv_1 = tf.nn.relu(conv2d(h_pool2, W_11_1) + b_11_1)

    # 3-3 conv after 1-1
    W_11_2 = weight_variable([1, 1, 64, 32],'W_11_2')
    b_11_2 = bias_variable([32])
    h_11_conv_2 = tf.nn.relu(conv2d(h_pool2, W_11_2) + b_11_2)

    W_33 = weight_variable([3, 3, 32, 64],'W_33')
    b_33 = bias_variable([64])
    h_33_conv = tf.nn.relu(tf.nn.conv2d(h_11_conv_2, W_33, strides=[1, 1, 1, 1], padding='SAME') + b_33)

    # 3-3 max pooling after 1-1 conv
    h_33 = tf.nn.max_pool(h_pool2, ksize=[1, 3, 3, 1], strides=[1, 1, 1, 1], padding='SAME')

    W_11_4 = weight_variable([1, 1, 64, 32],'W_11_4')
    b_11_4 = bias_variable([32])
    h_11_conv_4 = tf.nn.relu(conv2d(h_33, W_11_4) + b_11_4)

    # concat
    h_cat = tf.concat(3, [h_11_conv_1, h_33_conv, h_11_conv_4])
    
    # Add one more conv layer here to reduce paras
    W_conv3 = weight_variable([5, 5, 160, 256],'W_11_1')
    b_conv3 = bias_variable([256])
    h_conv3 = tf.nn.relu(conv2d(h_cat, W_conv3) + b_conv3)

    # fully connected layer, including 1024 nodes, too much!
    W_fc1 = tf.Variable(tf.random_uniform(shape=[256, 1024], minval=-0.1, maxval=0.1), name='W_fc1')
    b_fc1 = bias_variable([1024])

    h_conv3_flat = tf.reshape(h_conv3, [-1, 256])
    h_fc1 = tf.nn.relu(tf.matmul(h_conv3_flat, W_fc1) + b_fc1)

    # drop out layer
    h_fc1_drop = tf.nn.dropout(h_fc1, keep_prob)
    
    """
    After the fully connected layer, 5 seperate classifier was implemented. Then, 5 logits and loss was calculated,
    and the optimizer operates on the average of 5 loss, with exponential decay learning rate.
    """
    
     # implement 5 separate classifiers after a fully connected layer of 1024 inputs

    W_1 = tf.Variable(tf.random_uniform(shape=[1024, 11], minval=-0.1, maxval=0.1), name='W_1')
    b_1 = bias_variable([11])
    W_2 = tf.Variable(tf.random_uniform(shape=[1024, 11], minval=-0.1, maxval=0.1), name='W_2')
    b_2 = bias_variable([11])
    W_3 = tf.Variable(tf.random_uniform(shape=[1024, 11], minval=-0.1, maxval=0.1), name='W_3')
    b_3 = bias_variable([11])
    W_4 = tf.Variable(tf.random_uniform(shape=[1024, 11], minval=-0.1, maxval=0.1), name='W_4')
    b_4 = bias_variable([11])
    W_5 = tf.Variable(tf.random_uniform(shape=[1024, 11], minval=-0.1, maxval=0.1), name='W_5')
    b_5 = bias_variable([11])

    logits_1 = tf.matmul(h_fc1_drop, W_1) + b_1
    logits_2 = tf.matmul(h_fc1_drop, W_2) + b_2
    logits_3 = tf.matmul(h_fc1_drop, W_3) + b_3
    logits_4 = tf.matmul(h_fc1_drop, W_4) + b_4
    logits_5 = tf.matmul(h_fc1_drop, W_5) + b_5

    # Loss function
    loss_1 = 5 * tf.nn.softmax_cross_entropy_with_logits(logits_1, y_1)  # different weights on different classifiers
    loss_2 = 5 * tf.nn.softmax_cross_entropy_with_logits(logits_2, y_2)
    loss_3 = 3 * tf.nn.softmax_cross_entropy_with_logits(logits_3, y_3)
    loss_4 = tf.nn.softmax_cross_entropy_with_logits(logits_4, y_4)
    loss_5 = tf.nn.softmax_cross_entropy_with_logits(logits_5, y_5)

    # This tf.scalar_summary is served for tensorboard, to draw out decline of total_loss
    with tf.name_scope("loss") as scope:
        total_loss = tf.reshape((loss_1 + loss_2 + loss_3 + loss_4 + loss_5), [tf.shape(logits_1)[0], 1])
        tf.scalar_summary("total_loss", tf.reduce_mean(total_loss))

        prediction = tf.transpose(tf.pack(
            [tf.argmax(logits_1, 1),
             tf.argmax(logits_2, 1),
             tf.argmax(logits_3, 1),
             tf.argmax(logits_4, 1),
             tf.argmax(logits_5, 1)]))

    # The usage of exponential decay learning rate and AdagradOptimizer
    global_step = tf.Variable(0)
    learning_rate = tf.train.exponential_decay(0.01, global_step, 10000, 0.95)
    optimizer = tf.train.AdagradOptimizer(learning_rate).minimize(total_loss, global_step=global_step)
    tf.scalar_summary("learning_rate", learning_rate)


    def accuracy(pred2, batch_y2):
        """
        Calculate by counting if it's correct on every digits.
        """
        cnt = 0
        compare = tf.equal(pred2, tf.argmax(batch_y2, 2)).eval()
        for item in compare:
            cnt += (sum(item) / 5.0)
            # if sum(item)==5:
            #     cnt+=1
        return 100.0 * cnt / np.shape(batch_y2)[0]


    def foreign_evaluate(data_x=va_x, data_y=va_y):

        feed_dict = {x: b_x, y_1: b_y[:, 0], y_2: b_y[:, 1], y_3: b_y[:, 2],
